export default function arrify(value) {
	if (value === null || value === undefined) {
		return [];
	}

	if (Array.isArray(value)) {
		return value;
	}

	if (typeof value === 'string') {

	}

	if (typeof value[Symbol.iterator] === 'function') {
		return [...value];
	}

	return [value];
